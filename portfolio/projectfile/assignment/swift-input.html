Check leep year in Swift Programing Language

Code (Terminal Input)

import Foundation print("Enter a year: ", terminator: "") if let input = readLine(), let year = Int(input) { if (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0) { print("\(year) is a leap year.") } else { print("\(year) is not a leap year.") } } else { print("Invalid input. Please enter a valid year.") }

Output

Enter a year: 2024 2024 is a leap year.

Code (Assigned Value)

import Foundation let year = 2024 // Assigned value if (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0) { print("\(year) is a leap year.") } else { print("\(year) is not a leap year.") }
